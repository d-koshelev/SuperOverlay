private void RefreshEditorCanvasBounds()
{
    if (_session is null)
    {
        return;
    }

    var canvas = _session.GetCanvas();

    RootGrid.Width = canvas.Width;
    RootGrid.Height = canvas.Height;

    CanvasFrameBorder.Width = canvas.Width;
    CanvasFrameBorder.Height = canvas.Height;

    CanvasWidthTextBox.Text = ((int)Math.Round(canvas.Width)).ToString();
    CanvasHeightTextBox.Text = ((int)Math.Round(canvas.Height)).ToString();

    SyncCanvasPresetSelection(canvas.Width, canvas.Height);
}

private void ApplyCanvasSize_OnClick(object sender, WpfRoutedEventArgs e)
{
    try
    {
        if (!double.TryParse(CanvasWidthTextBox.Text, out var width) ||
            !double.TryParse(CanvasHeightTextBox.Text, out var height))
        {
            WpfMessageBox.Show(this, "Canvas width and height must be valid numbers.", "Apply Canvas");
            return;
        }

        _session?.SetCanvasSize(width, height);
        RefreshEditorCanvasBounds();
    }
    catch (Exception ex)
    {
        WpfMessageBox.Show(this, ex.Message, "Apply Canvas");
    }
}

private void CanvasPresetComboBox_OnSelectionChanged(object sender, WpfSelectionChangedEventArgs e)
{
    if (CanvasPresetComboBox.SelectedItem is not System.Windows.Controls.ComboBoxItem item)
    {
        return;
    }

    var tag = item.Tag?.ToString();
    if (string.IsNullOrWhiteSpace(tag))
    {
        return;
    }

    var parts = tag.Split('x');
    if (parts.Length != 2)
    {
        return;
    }

    if (double.TryParse(parts[0], out var width) &&
        double.TryParse(parts[1], out var height))
    {
        CanvasWidthTextBox.Text = width.ToString("0");
        CanvasHeightTextBox.Text = height.ToString("0");
    }
}

private void ApplyCanvasPreset_OnClick(object sender, WpfRoutedEventArgs e)
{
    ApplyCanvasSize_OnClick(sender, e);
}
